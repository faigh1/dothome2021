16. 재귀 함수

{
    function func(num) {
        if(num <= 1){
            document.write("함수가 실행되었습니다.");
        } else {
            document.write("함수가 실행되었습니다.");
            func(num-1);
        }
    }
    func(5);
}

17. 콜백 함수

{
    function funA(){
        document.write("함수A가 실행되었습니다.")
    }
    function funB(str){
        document.write("함수B가 실행되었습니다.")
        str();
    }
    funB(funA);
}

18. 콜백 함수 : 반복문

{
    function func(){
        document.write("함수가 실행되었습니다.")
    }
    function callback(str){
        for (let i=1; i<=5; i++){
            str(i);
        }
    }
    callback(func);
}

19. 콜백함수 : 동기/비동기

{
    function funA(){
        console.log("funA가 실행되었습니다.");
    }
    function funB(){
        console.log("funB가 실행되었습니다.");
    }
    funB();
    funA();

    //02
    function funC(){
        setTimeout(()=>{
            console.log("funC가 실행되었습니다.");
        }, 1000);
    }
    function funD(){
        console.log("funD가 실행되었습니다.");
    }
    funC();
    funD();

    //03
    function funE(callback){
        setTimeout(()=>{
            console.log("funE가 실행되었습니다.");
            callback();
        },2000);
    }
    function funF(){
        console.log("funF가 실행되었습니다.")
    }
    funE(funF);
}
