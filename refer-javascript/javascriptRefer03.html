19. 콜백함수 : 동기/비동기

{
    function funA(){
        console.log("funA가 실행되었습니다.");
    }
    function funB(){
        console.log("funB가 실행되었습니다.");
    }
    funB();
    funA();

    //02
    function funC(){
        setTimeout(()=>{
            console.log("funC가 실행되었습니다.");
        }, 1000);
    }
    function funD(){
        console.log("funD가 실행되었습니다.");
    }
    funC();
    funD();

    //03
    function funE(callback){
        setTimeout(()=>{
            console.log("funE가 실행되었습니다.");
            callback();
        },2000);
    }
    function funF(){
        console.log("funF가 실행되었습니다.")
    }
    funE(funF);
}

20. 콜백 함수 : 콜백지옥

            console.log('B');
            callback();
        },1000);
    }
    function C(callback){
        setTimeout(function(){
            console.log('C');
            callback();
        },1000);
    }
    function D(callback){
        setTimeout(function(){
            console.log('D');
        },1000);
    }
    A(function(){
        B(function(){
            C(function(){
                D();
            });
        });
    });
}

25. 함수 유형 : 함수와 객체(함수)를 이용한 형태

26. 함수 유형 : 객체 생성자 함수

{
    function func(num, name, job){
        this.num = num;
        this.name = name;
        this.job = job;

        this.result = function(){
            document.write(this.num + ". 내 이름은 " + this.name + "이며, 직업은 " + this.job + "입니다.");
        }
    }

    const info1 = new func("1", "송지", "웹 퍼블리셔"); //인스턴스 생성
    const info2 = new func("2", "지원", "프론트앤드 개발자"); //인스턴스 생성

    info1.result();
    info2.result();
}

30. 함수 유형 : 클래스 상속

{
    class study {
        constructor(num, name, job){
            this.num = num;
            this.name = name;
            this.job = job;
        }
        result(){
            document.write(this.num + ". 내 이름은 " + this.name + "이며, 직업은 " + this.job + "입니다." );
        }
    }
    class study2 extends study {
        constructor(num, name, job, age){
            super(num, name, job);
            this.age = age;
        }
        result2(){
            document.write(this.num + ". 내 이름은" + this.name + "이며, 직업은 " + this.job + "이며 나이는 "+ this.age +"살 입니다." );
        }
    }

    const info1 = new study("1", "송지", "웹퍼블리셔");
    const info2 = new study2("2", "지원", "프론트앤드 개발자","25");

    info1.result();
    info2.result();
    info2.result2();
}
