01. 1부터 100까지 출력하기 

{
    for ( let i=1; i<=100; i++ ){
            document.write(i);
    }
}

02. 1부터 100까지 출력하기 : 짝수만

{
    for ( let i=2; i<=100; i+=2 ){
        document.write(i);
    }
}

03. 1부터 100까지 출력하기 : 5의 배수만 

{
    for ( let i=1; i<=100; i++ ){
        if( i % 5 == 0 ) document.write(i);
    }
}

04. 1부터 100까지 출력하기 : 짝수만 빨간색, 홀수는 파란색 출력

{
    for( let i=1; i<=100; i++){
        if( i % 2 == 0){
            document.write("<span style='color : red'> "+ i +"</span>");
        } else {
            document.write("<span style='color : blue'> "+ i +"</span>");
        }
    }
}

05. 1부터 100까지 합 출력하기

{
    let sum= 0;

    for(let i=1; i<=100; i++) {
        sum += i;
        //sum = sum + i;
    }
    document.write(sum);
}

06. 구구단 출력하기

{
    for(var i=2; i<=9; i++){
        for(var j=2; j<=9; j++){
            var sum = i * j;
        document.write(i + " * " + j + " = " + sum);
        }
    }
    document.write();
}

07. 테이블 만들기1

{
    let num = 1;
    let table = "<table>";

    for(let i=1; i<=10; i++){
        table += "<tr>"
            for( let j=1; j<=10; j++ ){
                table += "<td>"+ num +"</td>";
                num++;
            }
            table += "</tr>";
    }

    table += "</table>";

    document.write(table);
}
