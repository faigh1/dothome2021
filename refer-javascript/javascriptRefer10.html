01. 요소 객체 : 클래스 메서드 : classList.add() / classList.remove()

클릭하면 모든 이미지에 클래스(active)를 추가합니다. 클릭하면 모든 이미지에 클래스(active)를 삭제합니다. 
{
    document.querySelector('.btn1-1').addEventListener('click', () => {
        document.querySelectorAll('.ex1>div').forEach(img => img.classList.add('active'));
    });

    document.querySelector('.btn1-2').addEventListener('click', () => {
        document.querySelectorAll('.ex1>div').forEach(img => img.classList.remove('active'));
    });
}

02. 요소 객체 : 클래스 메서드 : classList.toggle()

이미지를 클릭하면 클래스(active)를 추가/삭제합니다. 
{
    //모든 이미지 선택
    document.querySelectorAll('.ex2 div').forEach(el => {
        el.addEventListener('click', () => {
            document.querySelectorAll('.ex2 div').forEach(el2 => {
                el2.classList.toggle('active');
            });
        });
    });

    //내가 클릭한 이미지만 선택
    document.querySelectorAll('.ex2 div').forEach(el => {
        el.addEventListener('click', function () {
            this.classList.toggle('active');
        });
    });
}

03. 요소 객체 : 클래스 메서드 : classList.contains()

리셋 클릭하면 .active 클래스를 찾은 후에 클래스를 제거합니다. 
{
    //리셋버튼을 클릭하면 1,4번째 클래스 추가
    document.querySelector(".btn3-0").addEventListener("click", () => {
        document.querySelector(".ex3 > div:nth-child(2)").classList.add('active');
        document.querySelector(".ex3 > div:nth-child(4)").classList.add('active');
    });

    document.querySelector(".btn3-1").addEventListener("click", () => {
        document.querySelectorAll('.ex3 div').forEach(el => {
            if (el.classList.contains('active')) {
                el.classList.remove('active');
            }
        });
    });
}
